import pygame import random import sys # Initialize Pygame pygame.init() # Constants WIDTH, HEIGHT = 800, 600 BLOCK_SIZE = 20 FPS = 10 # Colors WHITE = (255, 255, 255) BLACK = (0, 0, 0) RED = (255, 0, 0) GREEN = (0, 255, 0) # Display screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Snake Game") # Font & Clock font = pygame.font.Font(None, 36) clock = pygame.time.Clock() class SnakeGame: def __init__(self): self.snake = [(200, 200), (220, 200), (240, 200)] self.direction = 'RIGHT' self.game_over = False self.apple = self.generate_apple() def generate_apple(self): """Apple will NOT spawn on snake body""" while True: pos = ( random.randint(0, WIDTH - BLOCK_SIZE) // BLOCK_SIZE * BLOCK_SIZE, random.randint(0, HEIGHT - BLOCK_SIZE) // BLOCK_SIZE * BLOCK_SIZE ) if pos not in self.snake: return pos def move_snake(self): head = self.snake[-1] if self.direction == 'UP': new_head = (head[0], head[1] - BLOCK_SIZE) elif self.direction == 'DOWN': new_head = (head[0], head[1] + BLOCK_SIZE) elif self.direction == 'LEFT': new_head = (head[0] - BLOCK_SIZE, head[1]) else: # RIGHT new_head = (head[0] + BLOCK_SIZE, head[1]) self.snake.append(new_head) if new_head == self.apple: self.apple = self.generate_apple() else: self.snake.pop(0) # Collision check if ( new_head[0] < 0 or new_head[0] >= WIDTH or new_head[1] < 0 or new_head[1] >= HEIGHT or new_head in self.snake[:-1] ): self.game_over = True def draw(self): screen.fill(BLACK) if self.game_over: text = font.render("Game Over! Press R to Restart", True, WHITE) screen.blit(text, (WIDTH // 2 - 180, HEIGHT // 2)) else: for block in self.snake: pygame.draw.rect(screen, GREEN, (*block, BLOCK_SIZE, BLOCK_SIZE)) pygame.draw.rect( screen, RED, (self.apple[0], self.apple[1], BLOCK_SIZE, BLOCK_SIZE) ) score = len(self.snake) - 3 score_text = font.render(f"Score: {score}", True, WHITE) screen.blit(score_text, (10, 10)) pygame.display.flip() def play(self): while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: if event.key == pygame.K_UP and self.direction != 'DOWN': self.direction = 'UP' elif event.key == pygame.K_DOWN and self.direction != 'UP': self.direction = 'DOWN' elif event.key == pygame.K_LEFT and self.direction != 'RIGHT': self.direction = 'LEFT' elif event.key == pygame.K_RIGHT and self.direction != 'LEFT': self.direction = 'RIGHT' elif event.key == pygame.K_r and self.game_over: self.__init__() if not self.game_over: self.move_snake() self.draw() clock.tick(FPS) if __name__ == "__main__": SnakeGame().play()
